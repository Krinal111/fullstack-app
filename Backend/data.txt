
CREATE TYPE user_role AS ENUM ('Customer', 'Vendor', 'Admin');

CREATE TABLE users (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150), -- Made optional since we're using phone for auth
    password TEXT NOT NULL,
    phone_number VARCHAR(20) UNIQUE NOT NULL, -- Made unique and required
    role user_role DEFAULT 'Customer',
    shop_name VARCHAR(200), -- For vendors only
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Create a trigger function to auto-update updated_at
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
   NEW.updated_at = NOW();
   RETURN NEW;
END;
$$ language 'plpgsql';

-- Attach the trigger to vendors table
CREATE TRIGGER update_vendors_updated_at
BEFORE UPDATE ON vendors
FOR EACH ROW
EXECUTE FUNCTION update_updated_at_column();

CREATE TABLE vendor_timings (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vendor_id UUID NOT NULL REFERENCES vendors(id) ON DELETE CASCADE,
  meal_type VARCHAR(50) NOT NULL CHECK (meal_type IN ('lunch', 'dinner')),
  price NUMERIC NOT NULL,
  start_time TIME NOT NULL,
  end_time TIME NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 3. Attach trigger to table
CREATE TRIGGER trigger_set_updated_at
BEFORE UPDATE ON vendor_timings
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();
